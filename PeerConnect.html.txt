            const skill = input.value.trim();
            
            if (skill && !userSkills.includes(skill)) {
                userSkills.push(skill);
                renderSkills();
                input.value = '';
            }
        }

        function removeSkill(skill) {
            userSkills = userSkills.filter(s => s !== skill);
            renderSkills();
        }

        function renderSkills() {
            const container = document.getElementById('skillsContainer');
            container.innerHTML = userSkills.map(skill => 
                `<div class="skill-tag">
                    ${skill}
                    <span class="skill-remove" onclick="removeSkill('${skill}')">Ã—</span>
                </div>`
            ).join('');
        }

        function createProfile(event) {
            event.preventDefault();
            
            // Validate required fields
            const requiredFields = ['fullName', 'collegeId', 'year', 'department'];
            for (let field of requiredFields) {
                if (!document.getElementById(field).value.trim()) {
                    showToast('Please fill in all required fields!');
                    return;
                }
            }
            
            const profile = {
                name: document.getElementById('fullName').value,
                collegeId: document.getElementById('collegeId').value,
                year: document.getElementById('year').value,
                department: document.getElementById('department').value,
                skills: [...userSkills],
                projectAreas: document.getElementById('projectAreas').value,
                bio: document.getElementById('bio').value,
                avatar: document.getElementById('fullName').value.split(' ').map(n => n[0]).join('').toUpperCase()
            };
            
            currentUser = profile;
            localStorage.setItem('peerConnectUser', JSON.stringify(profile));
            
            displayProfile(profile);
            updateStats();
            showToast('Profile created successfully!');
        }

        // Student Rendering
        function renderStudents() {
            const grid = document.getElementById('studentsGrid');
            const filteredStudents = getFilteredStudents();
            
            if (filteredStudents.length === 0) {
                grid.innerHTML = `
                    <div class="card" style="grid-column: 1 / -1; text-align: center; padding: 3rem;">
                        <i class="fas fa-users" style="font-size: 3rem; color: var(--medium-gray); margin-bottom: 1rem;"></i>
                        <h3>No students found</h3>
                        <p>Try adjusting your search filters.</p>
                    </div>
                `;
                return;
            }
            
            grid.innerHTML = filteredStudents.map(student => {
                const isConnected = connections.includes(student.id);
                const isPending = pendingRequests.includes(student.id);
                
                let buttonClass = 'connect-btn';
                let buttonText = '<i class="fas fa-plus"></i> Connect';
                
                if (isConnected) {
                    buttonClass += ' connected';
                    buttonText = '<i class="fas fa-check"></i> Connected';
                } else if (isPending) {
                    buttonClass += ' pending';
                    buttonText = '<i class="fas fa-clock"></i> Pending';
                }
                
                return `
                    <div class="student-card">
                        <div class="student-avatar">${student.avatar}</div>
                        <div class="student-info">
                            <h3>${student.name}</h3>
                            <div class="student-meta">
                                ${student.year} â€¢ ${student.department}
                            </div>
                            <div class="student-interests">
                                ${student.interests.map(interest => 
                                    `<span class="interest-tag">${interest}</span>`
                                ).join('')}
                            </div>
                            <p style="color: var(--medium-gray); font-size: 0.9rem; margin-bottom: 1rem;">
                                ${student.bio}
                            </p>
                            <button class="${buttonClass}" onclick="connectWithStudent(${student.id})" 
                                    ${isConnected || isPending ? 'disabled' : ''}>
                                ${buttonText}
                            </button>
                        </div>
                    </div>
                `;
            }).join('');
        }

        function getFilteredStudents() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            const departmentFilter = document.getElementById('departmentFilter').value;
            const yearFilter = document.getElementById('yearFilter').value;
            
            return students.filter(student => {
                const matchesSearch = !searchTerm || 
                    student.name.toLowerCase().includes(searchTerm) ||
                    student.interests.some(interest => 
                        interest.toLowerCase().includes(searchTerm)
                    );
                
                const matchesDepartment = !departmentFilter || 
                    student.department === departmentFilter;
                
                const matchesYear = !yearFilter || 
                    student.year === yearFilter;
                
                return matchesSearch && matchesDepartment && matchesYear;
            });
        }

        function filterStudents() {
            renderStudents();
        }

        // Connection Management
        function connectWithStudent(studentId) {
            if (!pendingRequests.includes(studentId) && !connections.includes(studentId)) {
                pendingRequests.push(studentId);
                localStorage.setItem('peerConnectPending', JSON.stringify(pendingRequests));
                
                // Simulate auto-acceptance after 2 seconds
                setTimeout(() => {
                    acceptConnection(studentId);
                }, 2000);
                
                renderStudents();
                updateStats();
                showToast('Connection request sent!');
            }
        }

        function acceptConnection(studentId) {
            const index = pendingRequests.indexOf(studentId);
            if (index > -1) {
                pendingRequests.splice(index, 1);
                connections.push(studentId);
                
                localStorage.setItem('peerConnectPending', JSON.stringify(pendingRequests));
                localStorage.setItem('peerConnectConnections', JSON.stringify(connections));
                
                renderStudents();
                updateStats();
                showToast('Connection accepted! ðŸŽ‰');
            }
        }

        // Statistics
        function updateStats() {
            document.getElementById('totalConnections').textContent = connections.length;
            document.getElementById('pendingRequests').textContent = pendingRequests.length;
            
            // Calculate shared interests
            let sharedInterests = 0;
            if (currentUser && currentUser.skills) {
                const connectedStudents = students.filter(student => 
                    connections.includes(student.id)
                );
                
                connectedStudents.forEach(student => {
                    const commonInterests = student.interests.filter(interest =>
                        currentUser.skills.some(skill => 
                            skill.toLowerCase().includes(interest.toLowerCase()) ||
                            interest.toLowerCase().includes(skill.toLowerCase())
                        )
                    );
                    sharedInterests += commonInterests.length;
                });
            }
            
            document.getElementById('sharedInterests').textContent = sharedInterests;
        }

        // Data Persistence
        function loadUserData() {
            const savedUser = localStorage.getItem('peerConnectUser');
            const savedConnections = localStorage.getItem('peerConnectConnections');
            const savedPending = localStorage.getItem('peerConnectPending');
            
            if (savedUser) {

            if (savedConnections) {

            if (savedPending) {
                pendingRequests = JSON.parse(savedPending);
            }
        }

        // Enhanced interactivity
        document.addEventListener('DOMContentLoaded', function() {
            // Add smooth scrolling
            document.documentElement.style.scrollBehavior = 'smooth';
            
            // Add intersection observer for animations
            const observerOptions = {
                threshold: 0.1,
                rootMargin: '0px 0px -50px 0px'
            };
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.style.opacity = '1';
                        entry.target.style.transform = 'translateY(0)';
                    }
                });
            }, observerOptions);
            
            // Observe all cards for animations
            setTimeout(() => {
                document.querySelectorAll('.card, .student-card, .stat-card').forEach(card => {
                    card.style.opacity = '0';
                    card.style.transform = 'translateY(20px)';
                    card.style.transition = 'opacity 0.6s ease, transform 0.6s ease';
                    observer.observe(card);
                });
            }, 100);
        });
